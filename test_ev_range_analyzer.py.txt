import pytest
from unittest.mock import patch, MagicMock

from ev_range_analyzer import (
    get_location,
    get_elevation,
    get_weather,
    get_optimal_ev_speed,
    get_tomtom_flow,
    EVAnalyzer,
)


@patch("ev_range_analyzer.Nominatim")
def test_get_location_success(mock_nominatim):
    mock_geo = MagicMock()
    mock_geo.geocode.return_value = MagicMock(
        latitude=43.0,
        longitude=-81.0,
        address="London, Ontario, Canada",
        raw={"boundingbox": [1, 2, 3, 4]},
    )
    mock_nominatim.return_value = mock_geo

    result = get_location("London", "Canada")

    assert result["lat"] == 43.0
    assert result["lon"] == -81.0
    assert result["address"] == "London, Ontario, Canada"


@patch("ev_range_analyzer.Nominatim")
def test_get_location_not_found(mock_nominatim):
    mock_geo = MagicMock()
    mock_geo.geocode.return_value = None
    mock_nominatim.return_value = mock_geo

    assert get_location("FakeCity", "Nowhere") is None


@patch("ev_range_analyzer.requests.post")
def test_get_elevation_success(mock_post):
    mock_response = MagicMock()
    mock_response.json.return_value = {
        "results": [{"elevation": 250}]
    }
    mock_response.raise_for_status.return_value = None
    mock_post.return_value = mock_response

    elevation = get_elevation(43.0, -81.0)

    assert elevation == 250


@patch("ev_range_analyzer.requests.post")
def test_get_elevation_failure(mock_post):
    mock_post.side_effect = Exception("API down")

    assert get_elevation(43.0, -81.0) is None

@patch("ev_range_analyzer.requests.get")
def test_get_tomtom_flow(mock_get):
    mock_response = MagicMock()
    mock_response.status_code = 200
    mock_response.json.return_value = {
        "flowSegmentData": {
            "currentSpeed": 100,
            "freeFlowSpeed": 80,
            "currentTravelTime": 120,
            "freeFlowTravelTime": 100,
            "frc": "FRC1",
        }
    }
    mock_get.return_value = mock_response

    result = get_tomtom_flow(43.0, -81.0)

    curr, opt, delay, impact = result

    assert curr == 100
    assert opt == 90
    assert delay == 20
    assert impact in ["DRAG LOSS", "OPTIMAL", "IDLE LOSS"]

def test_temperature_analysis_cold():
    analyzer = EVAnalyzer({"temperature_avg": -15}, 100, [], {"congestion_ratio": 1})
    result = analyzer.analyze_temperature()

    assert result["score"] == 2


def test_elevation_analysis_high():
    analyzer = EVAnalyzer({}, 2500, [], {"congestion_ratio": 1})
    result = analyzer.analyze_elevation()

    assert result["score"] == 3


def test_infrastructure_low():
    analyzer = EVAnalyzer({}, 100, ["s1", "s2"], {"congestion_ratio": 1})

def test_traffic_analysis_moderate():
    analyzer = EVAnalyzer({}, 100, [], {"congestion_ratio": 1.8})
    result = analyzer.analyze_traffic()

    assert result["score"] == 7


def test_overall_score():
    analyzer = EVAnalyzer(
        {"temperature_avg": 10},
        500,
        ["s1", "s2", "s3", "s4", "s5"],
        {"congestion_ratio": 1.2},
    )

    avg, factors = analyzer.overall_score()

    assert len(factors) == 4
    assert 0 <= avg <= 10


def test_report_output(capsys):
    analyzer = EVAnalyzer(
        {"temperature_avg": 10},
        500,
        ["s1"] * 10,
        {"congestion_ratio": 1.2},
    )

    analyzer.report()

    captured = capsys.readouterr()

    assert "EV Suitability Report" in captured.out
    assert "Overall Score" in captured.out
